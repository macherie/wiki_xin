====== 音频参数文件 ======


===== acdb file在代码里位置: =====


cd vendor/qcom/proprietary/mm-audio/audcal/family-b/acdbdata/8226/

===== acdb file在手机里位置: =====


#adb shell

#/etc/acdbdata


===== 用以下命令可以把文件导入手机里： =====


adb shell setprop service.adb.root 1

adb remount && adb push Bluetooth_cal.acdb /etc/acdbdata/MTP/MTP_Bluetooth_cal.acdb && adb push General_cal.acdb /etc/acdbdata/MTP/MTP_General_cal.acdb && adb push Global_cal.acdb /etc/acdbdata/MTP/MTP_Global_cal.acdb && adb push Handset_cal.acdb /etc/acdbdata/MTP/MTP_Handset_cal.acdb && adb push Hdmi_cal.acdb /etc/acdbdata/MTP/MTP_Hdmi_cal.acdb && adb push Headset_cal.acdb /etc/acdbdata/MTP/MTP_Headset_cal.acdb && adb push Speaker_cal.acdb /etc/acdbdata/MTP/MTP_Speaker_cal.acdb && adb push Bluetooth_cal.acdb /etc/acdbdata/QRD/QRD_Bluetooth_cal.acdb && adb push General_cal.acdb /etc/acdbdata/QRD/QRD_General_cal.acdb && adb push Global_cal.acdb /etc/acdbdata/QRD/QRD_Global_cal.acdb && adb push Handset_cal.acdb /etc/acdbdata/QRD/QRD_Handset_cal.acdb && adb push Hdmi_cal.acdb /etc/acdbdata/QRD/QRD_Hdmi_cal.acdb && adb push Headset_cal.acdb /etc/acdbdata/QRD/QRD_Headset_cal.acdb && adb push Speaker_cal.acdb /etc/acdbdata/QRD/QRD_Speaker_cal.acdb && adb reboot
