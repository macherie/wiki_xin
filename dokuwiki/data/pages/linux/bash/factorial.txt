**如何在shell脚本里实现递归呢？请看下面的例子**

**使用递归求阶乘**

factorial()
{
        if [ $1 -eq 1 ] 
        then
                echo 1
        else
                local tmp=$[ $1 - 1 ]
                local ret=`factorial $tmp`
                echo $[ $ret * $1 ]
        fi
}

read -p "enter value:" value

ret=`factorial $value`

echo "the factorial of $value is $ret"
